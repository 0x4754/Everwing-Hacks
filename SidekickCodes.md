		}
	},
	...

// This will print them nice
var sidekickcodes_md = "", names;

	sidekickcodes_md += " - **" + i + "** : " + names.tier0 + ", " + names.tier1 + ", " + names.tier2 + "\n";
}
console.log(sidekickcodes_md);

}
